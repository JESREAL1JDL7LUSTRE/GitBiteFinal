import { getReviews } from "@/api/DishApi";
import { ReviewsItem } from "@/types/Types";
import { useState, useEffect } from "react";

const useFetchReviews = (dish_id: number | null) => {
  const [reviews, setReviews] = useState<ReviewsItem[]>([]);
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!dish_id) return;

    const fetchReviews = async () => {
      setLoading(true);
      try {
        const data = await getReviews(dish_id);
        setReviews(data);
      } catch {
        setError("Failed to fetch reviews");
      } finally {
        setLoading(false);
      }
    };

    fetchReviews();
  }, [dish_id]); // Re-fetch when dish_id changes

  return { reviews, loading, error };
};

export default useFetchReviews;

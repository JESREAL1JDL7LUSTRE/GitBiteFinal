import { createOrder } from "@/api/OrderApi";
import { Order } from "@/types/Types";
import { useState } from "react";
import { useNavigate } from "react-router-dom";

const usePostOrder = () => {
  const [order, setOrder] = useState<Order | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const nav = useNavigate();

  const CreateOrder = async (dishDetails: { id: number; quantity: number }[]) => {
    setLoading(true);
    setError(null);

    try {
      const newOrder = await createOrder(dishDetails);
      setOrder(newOrder);
      return newOrder;
    } catch (error) {
      setError(`Error creating order. Please try logging in. ${error}`);
      alert("Error adding items to Order. Please log in.");
      nav("/signin");
    } finally {
      setLoading(false);
    }
  };

  return { CreateOrder, order, loading, error };
};

export default usePostOrder;

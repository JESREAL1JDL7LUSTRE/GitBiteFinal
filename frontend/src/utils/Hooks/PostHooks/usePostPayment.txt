import { postPayment } from "@/api/PaymentApi";
import { PaymentData } from "@/types/Types";
import { useState } from "react";

const usePostPayment = () => {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const PostPayment = async (paymentData: PaymentData) => {
    setLoading(true);
    setError(null);

    try {
      await postPayment(paymentData);
    } catch {
      setError("Payment failed. Please try again.");
    } finally {
      setLoading(false);
    }
  };

  return { PostPayment, loading, error };
};

export default usePostPayment;

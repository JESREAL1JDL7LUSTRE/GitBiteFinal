import { changePassword } from "@/api/ChangePassApi";
import { useState } from "react";

const usePostChangePassword = () => {
  const [message, setMessage] = useState<string | null>(null);
  const [loading, setLoading] = useState<boolean>(false);

  const handleChangePassword = async (oldPassword: string, newPassword: string) => {
    setLoading(true);
    setMessage(null);

    try {
      const successMessage = await changePassword(oldPassword, newPassword);
      setMessage(successMessage);
    } catch (errorMessage) {
      setMessage(errorMessage as string);
    } finally {
      setLoading(false);
    }
  };

  return { message, loading, changePassword: handleChangePassword };
};

export default usePostChangePassword;
